--Vyberte prvních 5 výrobcù (tabulka manufacturers) seøazených podle názvu abecednì vzestupnì.
SELECT*FROM SQL1.MANUFACTURERS
ORDER BY manufacturer asc

--Zjistìte všechny produkty z kategorie Youth a seøaï je podle ceny (priceNew) sestupnì.
SELECT* FROM SQL1.PRODUCTS
WHERE Category = 'Youth'
ORDER BY PRICENEW desc;
